def run_regression2(X, Y, lam):
    X = np.hstack(([1],X))
    d = 2
    I = np.identity(d)
    Y = np.array(Y)
    beta = np.linalg.inv(X.T.dot(X) + lam * I).dot(X.T).dot(Y)
    return beta

def fit_linear_regression(data_matrix, response_list):
    """
    :param data_matrix: A matrix, where each row is a data element (X)
    :param response_list: A list of responses for each of the rows (y)
    :return: A vector containing the hyperplane equation (beta)
    """
    X = np.array([[1, *x] for x in data_matrix])
    Y = np.array(response_list)
    beta = np.linalg.inv(X.T.dot(X)).dot(X.T).dot(Y)
    return beta

def fit_ridge_polynomial_regression(data_matrix, response_list, degree=2, l=0.1):
    """
    :param data_matrix: A matrix, where each row is a data element (X)
    :param response_list: A list of responses for each of the rows (y)
    :param degree: degree of polynomial to use
    :param l: ridge variable
    :return: A vector containing the hyperplane equation (beta)
    """
    X = np.array([[x[0] ** i for i in range(0, n + 1)] for x in data_matrix])
    Y = np.array(response_list)
    D = X.shape[1]  # dimension + 1
    beta = np.linalg.inv(X.T.dot(X) + l * np.identity(D)).dot(X.T).dot(Y)
    return beta

df = pd.read_csv('presentation_data/lec2/center.csv')

X = [[apt['area']] for index, apt in df.iterrows()]  # Note that X is a matrix.
# Divide price by 1000, so it's more readable.
Y = [apt['price'] / 1000 for index, apt in df.iterrows()]

print('degree, poly_sq_err, ridge_sq_err')
for n in range(1, 10):
    curve_x = np.arange(20, 160, 1)
    parts = 5
    r_errors, p_errors = [], []
    for k in range(parts):
        test_X = X[k::parts]
        test_Y = Y[k::parts]
        train_X = np.array([X[j] for j in range(len(X)) if (j % parts) != k])
        train_Y = np.array([Y[j] for j in range(len(Y)) if (j % parts) != k])
        beta = fit_ridge_polynomial_regression(train_X, train_Y, degree=n, l=1)
        curve_y = [sum(beta[i] * x ** i for i in range(len(beta))) for x in curve_x]
        r_errors.append(sum((y - sum(beta[i] * x[0] ** i for i in range(len(beta)))) ** 2
                          for x, y in zip(test_X, test_Y)) / len(test_Y))
        beta = fit_polynomial_regression([x for x in X], Y, degree=n)
        curve_y = [sum(beta[i] * x ** i for i in range(len(beta))) for x in curve_x]
        p_errors.append(sum((y - sum(beta[i] * x[0] ** i for i in range(len(beta)))) ** 2
                        for x, y in zip(test_X, test_Y)) / len(test_Y))
    print('{:6d}  {:10,.1f}   {:,.1f}'.format(n, np.mean(r_errors), np.mean(p_errors)))
